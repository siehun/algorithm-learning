        }

        int result=stack.pop();
        if(stack.isEmpty()){
            return result;
        }
        else{
            int last=f(stack);
            stack.push(result);
            return last;
        }

            res.add(String.valueOf(str));

                visit[str[j]-'a']=true;
                swap(str,i,j);
                process2(str,i+1,res);
                swap(str,i,j);
            }
        }
    }

            return -1;

            return 0;

            return 0;
        }
        int N=arr.length;
        int[][] f=new int[N][N];
        int[][] s=new int[N][N];

            f[i][i]=arr[i];
        }
        for(int i=1;i<N;i++){

                s[row][col]=Math.min(f[row+1][col],f[row][col-1]);
                row++;
                col++;
            }
        }
        return Math.max(f[0][N-1],s[0][N-1]);
    }

    public static int process1(int i,int[] record,int n){
        if(i==n){
            return 1;
        }
        int res=0;
        for(int j=0;j<n;j++){
            if(isValid(record,i,j)){
                record[i]=j;
                res+=process1(i+1,record,n);
            }
        }
        return res;
    }
    public static boolean isValid(int[] record,int i,int j){
        for(int k=0;k<i;k++){
            if(j==record[k]||Math.abs(record[k]-j)==Math.abs(i-k)){
                return false;
            }
        }
        return true;
    }

    public static int num2(int n){
        if(n < 1 || n > 32){
            return 0;
        }
        int limit=n==32?-1:(1<<n)-1;
        return process2(limit,0,0,0);

    public static int process2(int limit,int colLim,int leftDiaLim,int rightDiaLim){
        if(colLim==limit){
            return 1;
        }
        int pos=limit & (~(colLim|leftDiaLim|rightDiaLim));
        int mostRightOne = 0;
        int res=0;
        while(pos!=0){
            mostRightOne=pos & (~pos + 1);
            pos=pos-mostRightOne;
            res+=process2(limit,colLim|mostRightOne,(leftDiaLim|mostRightOne)<<1,                  
            (rightDiaLim|mostRightOne)>>>1);
        }
        return res;
    }

    public static int way1(int N,int M,int K,int P){
        if(N < 2 || K < 1 || M < 1 || M > N || P < 1 || P > N){
            return 0;
        }
        return walk(N, M, K, P);
    }
    public static int walk(int N,int cur,int rest,int P){
        if(rest==0){
            return cur== P ? 1:0;
        }
        if(cur==1){
            return walk(N,2,rest-1,P);
        }
        if(cur == N){
            return walk(N, N-1, rest-1, P);
        }
        return walk(N, cur+1, rest-1, P) + walk(N, cur-1, rest-1, P);
    }
    public static int way2(int N,int M,int K,int P){
        if(N < 2 || K < 1 || M < 1 || M > N || P < 1 || P > N){
            return 0;
        }
        int[][] dp=new int[N+1][K+1];
        for(int row=0; row <= N; row++){
            for(int col=0; col <= K; col++){
                dp[row][col]=-1;
            }
        }
        return walk1(N, M, K, P,dp);
    }
    public static int walk1(int N,int cur,int rest,int P,int[][] dp){
        if(dp[cur][rest]!=-1){
            return dp[cur][rest];
        }
        if(rest==0){
            dp[cur][rest]=cur== P ? 1:0;
            return dp[cur][rest];
        }
        if(cur==1){
            dp[cur][rest]=walk1(N,2,rest-1,P,dp);
            return dp[cur][rest];
        }
        if(cur == N){
            dp[cur][rest]=walk1(N, N-1, rest-1, P,dp);
            return dp[cur][rest];
        }
        dp[cur][rest]= walk1(N, cur+1, rest-1, P,dp) + walk1(N, cur-1, rest-1, P,dp);
        return dp[cur][rest];
    }

    public static int minStickers1(String[] stickers, String target){
        int n = stickers.length;
        int[][] map=new int[n][26];
        //dp表示的是解决剩余的字符串的方法数
        HashMap<String,Integer> dp=new HashMap<>();
        //map与stickers其实是等效的，储存可选择的贴纸信息
        for(int i=0;i<n;i++){
            char[] str= stickers[i].toCharArray();
            for(char c:str){
                map[i][c-'a']++;
            }
        }
        //空字符串，0种方法
        dp.put("",0);
        return process1(dp,map,target);
    }
    public static int process1(HashMap<String,Integer> dp,int[][] map,String rest){
        // 如果当前这种情况已经计算过，那么直接返回
        if(dp.containsKey(rest)){
            return dp.get(rest);
        }
        //假设答案无效
        int ans = Integer.MAX_VALUE;
        int n = map.length;
        //tmp存储关于rest的信息
        int[] tmap=new int[26];
        //target存储rest
        char[] target=rest.toCharArray();
        for(char c:target){
            tmap[c - 'a']++;
        }
        for(int i = 0; i < n; i++){
            // map[i]代表的是可选择的第i个字符串，后面的26位代表的是每个字符的数量
            //target[0]-'a'去试目标字符串的第一个字符在map[i]中是否存在，如果存在
            if(map[i][target[0]-'a']==0){
                continue;
            }
            StringBuilder sb = new StringBuilder();
            for(int j = 0; j < 26; j++){
                if(tmap[j] > 0){
                    // 选择完贴纸后减一轮数量，把减完后的样子扔到sb里面
                    for(int k = 0; k < Math.max(0, tmap[j] - map[i][j]); k++){
                        sb.append((char)('a' + j));
                    }
                }
            }
            String s = sb.toString();
            //下一轮递归
            int tmp = process1(dp, map, s);
            //如果答案有效的话更新ans
            if(tmp != -1){
                ans = Math.min(ans, 1 + tmp);
            }
        }
        dp.put(rest, ans == Integer.MAX_VALUE ? -1 : ans);
        return dp.get(rest);
    }
